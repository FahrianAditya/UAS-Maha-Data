3. Cross Validation

RNGkind(sample.kind = "Rounding")
set.seed(100)
splitter <- initial_split(uas_svm, prop = 0.7, strata = "Pelunasan Kredit")

data_train <- training(splitter)
data_test <- testing(splitter)

test <- data_test %>% 
  select(-`Pelunasan Kredit`) ## mengeluarkan variabel target
